create table copasa(
	COPASA_ID int primary KEY auto_increment,
    LOCALIDADE VARCHAR (100),
    PROJETO VARCHAR (100),
    PROG_ORCAMENTARIO VARCHAR(100),
    REQUISICAO VARCHAR(100),
    DIAGRAMA_DE_REDE VARCHAR(100),
    TEXTO_DIAGRAMA VARCHAR(100),
    ANALISTA_FISCAL VARCHAR(100),
    PLANEJADO float,
    COMPROMISSO FLOAT,
    TOTAL_MEDIDO FLOAT,
    MEDIDO_REAJUSTE FLOAT,
    SALDO FLOAT,
    STATUS VARCHAR(50),
    DATA_STATUS date,
    DESVIO FLOAT,
    COORDERNADORES VARCHAR (200),
	PORCENTAGEM_MEDIDA int,
    OBSERVACAO VARCHAR(200)
    )
